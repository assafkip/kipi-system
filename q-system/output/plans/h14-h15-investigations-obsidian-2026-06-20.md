## H14 -- new file: `investigations/export/bases.py`

Copy this file verbatim into `investigations/export/bases.py`:

```python
"""Obsidian Bases (.base) exporter for the investigation vault. (H14)

An Obsidian `.base` file is a declarative YAML view over note frontmatter --
no backend, no database, just a filtered/sorted projection of the entity notes
that obsidian.py already writes. This pairs with the H15 frontmatter enrichment
(threat_score / threat_tier / role) so the views below can filter and sort on
real fields.

Output: vault/entities.base -- open it in Obsidian 1.7+ for table/cards views.
Signature mirrors canvas.export / obsidian.export (conn, vault_dir) so the
export orchestrator can register it the same way; conn is accepted for parity.
"""
from pathlib import Path


# Threat tiers must match the H15 enrichment in obsidian.py and the height
# thresholds in canvas.py (>=70 high, >=50 medium) so the three exporters agree.
def threat_tier(score):
    s = score or 0
    if s >= 70:
        return "high"
    if s >= 50:
        return "medium"
    return "low"


def _render_entities_base() -> str:
    return (
        "# Obsidian Bases view over the entity notes obsidian.py writes.\n"
        "# Generated by investigations/export/bases.py (H14). Do not hand-edit;\n"
        "# re-run the exporter. Requires the H15 frontmatter fields\n"
        "# (threat_score, threat_tier, role) on each entity note.\n"
        "filters:\n"
        "  and:\n"
        '    - \'file.hasTag("entity")\'\n'
        "properties:\n"
        "  threat_score:\n"
        "    displayName: Threat\n"
        "  threat_tier:\n"
        "    displayName: Tier\n"
        "  role:\n"
        "    displayName: Role\n"
        "  type:\n"
        "    displayName: Type\n"
        "views:\n"
        "  - type: table\n"
        "    name: All entities\n"
        "    order:\n"
        "      - file.name\n"
        "      - type\n"
        "      - role\n"
        "      - threat_tier\n"
        "      - threat_score\n"
        "    sort:\n"
        "      - property: threat_score\n"
        "        direction: DESC\n"
        "  - type: table\n"
        "    name: High threat\n"
        "    filters:\n"
        "      and:\n"
        '        - \'threat_tier == "high"\'\n'
        "    order:\n"
        "      - file.name\n"
        "      - role\n"
        "      - threat_score\n"
        "    sort:\n"
        "      - property: threat_score\n"
        "        direction: DESC\n"
        "  - type: cards\n"
        "    name: By type\n"
        "    order:\n"
        "      - file.name\n"
        "      - threat_score\n"
    )


def export(conn, vault_dir) -> Path:
    vault_dir = Path(vault_dir)
    vault_dir.mkdir(parents=True, exist_ok=True)
    base_path = vault_dir / "entities.base"
    base_path.write_text(_render_entities_base(), encoding="utf-8")
    return base_path
```

```python
def _entity_role(notes):
    if not notes:
        return ""
    return (notes or "").split(" — ")[0].replace("role:", "").strip()


def _threat_tier(score):
    s = score or 0
    if s >= 70:
        return "high"
    if s >= 50:
        return "medium"
    return "low"


_THREAT_CALLOUT = {"high": "danger", "medium": "warning", "low": "info"}
_THREAT_LABEL = {"high": "High threat", "medium": "Medium threat", "low": "Low threat"}
```

**Edit 3 -- enrich `_render_entity_md`** signature, frontmatter, and body callout.
Change the signature to accept `score`:

```python
def _render_entity_md(entity, mentions: list[dict], rels: list[dict],
                      aliases: list[str], score: dict | None = None) -> str:
    role = _entity_role(entity["notes"])
    threat_score = int((score or {}).get("threat_score", 0) or 0)
    tier = _threat_tier(threat_score)
    lines = [
        "---",
        f"name: {entity['canonical_name']}",
        f"type: {entity['entity_type']}",
        f"role: {role or 'unknown'}",
        f"threat_score: {threat_score}",
        f"threat_tier: {tier}",
        f"tags: [entity, {entity['entity_type']}]",
    ]
```

(That replaces the old 4-line `lines = ["---", name, type, tags]` opener.)
Then, immediately after the existing `lines.append(f"**Type:** {entity['entity_type']}")`,
add the H15 callout:

